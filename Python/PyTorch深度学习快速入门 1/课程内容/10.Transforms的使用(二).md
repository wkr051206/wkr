```python
from PIL import Image  # 导入PIL的Image模块，用于打开和处理图像
from torchvision import transforms  # 导入PyTorch的图像转换模块
img_path = "dataset/train/ants_image/0013035.jpg"  # 相对路径指定图像位置
img = Image.open(img_path)  # 使用PIL打开图像文件
tensor_trans = transforms.ToTensor()  # 创建ToTensor转换器实例
tensor_img = tensor_trans(img)  # 将PIL图像转换为PyTorch张量
print(tensor_img)  # 打印转换后的张量信息
```
